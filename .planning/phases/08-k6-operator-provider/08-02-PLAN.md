   ```go
   func (p *K6OperatorProvider) TriggerRun(ctx context.Context, cfg *provider.PluginConfig) (string, error) {
       // 1. Validate config
       if err := cfg.ValidateK6Operator(); err != nil {
           return "", err
       }

       // 2. Read script from ConfigMap (proves ConfigMap exists)
       _, err := p.readScript(ctx, cfg)
       if err != nil {
           return "", err
       }

       // 3. Get clients
       _, dynClient, err := p.ensureClient()
       if err != nil {
           return "", err
       }

       // 4. Resolve namespace
       ns := cfg.Namespace
       if ns == "" {
           ns = "default"
       }

       // 5. Generate CR name
       rolloutName := cfg.RolloutName
       if rolloutName == "" {
           rolloutName = "unknown"
       }
       crName := testRunName(rolloutName, ns)

       // 6. Build and create CR (auto-detect TestRun vs PrivateLoadZone per D-02)
       if isCloudConnected(cfg) {
           plz := buildPrivateLoadZone(cfg, cfg.ConfigMapRef.Name, cfg.ConfigMapRef.Key, ns, crName, rolloutName)
           obj, err := runtime.DefaultUnstructuredConverter.ToUnstructured(plz)
           if err != nil {
               return "", fmt.Errorf("convert PrivateLoadZone to unstructured: %w", err)
           }
           u := &unstructured.Unstructured{Object: obj}
           created, err := dynClient.Resource(plzGVR).Namespace(ns).Create(ctx, u, metav1.CreateOptions{})
           if err != nil {
               return "", fmt.Errorf("create PrivateLoadZone %s/%s: %w", ns, crName, err)
           }
           slog.Info("k6 PrivateLoadZone created",
               "name", created.GetName(),
               "namespace", ns,
               "provider", p.Name(),
           )
           return created.GetName(), nil
       }

       tr := buildTestRun(cfg, cfg.ConfigMapRef.Name, cfg.ConfigMapRef.Key, ns, crName, rolloutName)
       obj, err := runtime.DefaultUnstructuredConverter.ToUnstructured(tr)
       if err != nil {
           return "", fmt.Errorf("convert TestRun to unstructured: %w", err)
       }
       u := &unstructured.Unstructured{Object: obj}
       created, err := dynClient.Resource(testRunGVR).Namespace(ns).Create(ctx, u, metav1.CreateOptions{})
       if err != nil {
           return "", fmt.Errorf("create TestRun %s/%s: %w", ns, crName, err)
       }
       slog.Info("k6 TestRun created",
           "name", created.GetName(),
           "namespace", ns,
           "provider", p.Name(),
       )
       return created.GetName(), nil
   }
   ```

6. **Replace GetRunResult stub:**
   Follow the cloud.go GetRunResult pattern:
   ```go
   func (p *K6OperatorProvider) GetRunResult(ctx context.Context, cfg *provider.PluginConfig, runID string) (*provider.RunResult, error) {
       client, dynClient, err := p.ensureClient()
       if err != nil {
           return nil, err
       }

       ns := cfg.Namespace
       if ns == "" {
           ns = "default"
       }

       // Determine which GVR to use (per D-02)
       gvr := testRunGVR
       if isCloudConnected(cfg) {
           gvr = plzGVR
       }

       // Single GET per D-04 (no polling loop)
       u, err := dynClient.Resource(gvr).Namespace(ns).Get(ctx, runID, metav1.GetOptions{})
       if err != nil {
           return nil, fmt.Errorf("get TestRun %s/%s: %w", ns, runID, err)
       }

       // Extract stage from status
       stage, _, err := unstructured.NestedString(u.Object, "status", "stage")
       if err != nil {
           return nil, fmt.Errorf("extract stage from TestRun %s/%s: %w", ns, runID, err)
       }

       // Map stage to RunState
       state := stageToRunState(stage)

       slog.Debug("TestRun status polled",
           "name", runID,
           "namespace", ns,
           "stage", stage,
           "state", state,
           "provider", p.Name(),
       )

       // For terminal stages, check runner pod exit codes (per D-05, issue #577 workaround)
       if stage == "finished" {
           exitState, err := checkRunnerExitCodes(ctx, client, ns, runID)
           if err != nil {
               slog.Warn("failed to check runner exit codes, treating as error",
                   "name", runID,
                   "error", err,
                   "provider", p.Name(),
               )
               return &provider.RunResult{
                   State: provider.Errored,
               }, nil
           }
           state = exitState
       }

       result := &provider.RunResult{
           State:            state,
           ThresholdsPassed: state == provider.Passed,
       }
       return result, nil
   }
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Plugin -> k8s API server | CRD Create/Get/Delete calls cross process boundary to kube-apiserver |
| k8s API server -> Plugin | TestRun status and pod exit codes flow back into pass/fail logic |
| User config -> CRD spec | PluginConfig fields (image, env, args) flow into TestRun CR submitted to cluster |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-05 | Spoofing | operator.go / GetRunResult | accept | Dynamic client authenticates via in-cluster service account token. Spoofing kube-apiserver requires cluster-level compromise which is out of scope. |
| T-08-06 | Tampering | operator.go / TriggerRun | mitigate | Use `runtime.DefaultUnstructuredConverter` for typed->unstructured conversion (not manual map construction) to prevent field injection. TypeMeta always set explicitly before conversion per Pitfall 1. |
| T-08-07 | Denial of Service | operator.go / TriggerRun | mitigate | Each TriggerRun creates exactly one CR. The controller's reconcile cadence limits creation rate. Parallelism is validated in Plan 01 (non-negative). Resource limits on runner pods are user-configured. |
| T-08-08 | Repudiation | operator.go / all methods | mitigate | All CRD operations are logged via slog.Info with name, namespace, and provider. Kubernetes audit log also captures API server interactions. |
| T-08-09 | Elevation of Privilege | operator.go / StopRun | accept | Delete permission is granted by RBAC. Plugin can only delete resources its service account has access to. Over-permissive RBAC is a cluster admin concern, not a plugin concern. |
</threat_model>
