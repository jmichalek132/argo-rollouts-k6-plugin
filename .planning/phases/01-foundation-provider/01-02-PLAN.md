Per research finding #7 -- MagicCookieKey is `ARGO_ROLLOUTS_RPC_PLUGIN`, MagicCookieValue for metric plugin is `"metricprovider"`:

```go
package main

import (
    "log/slog"
    "os"
    "strings"

    goPlugin "github.com/hashicorp/go-plugin"
)

// handshakeConfig must match the argo-rollouts controller's metric plugin client.
// Source: github.com/argoproj/argo-rollouts/metricproviders/plugin/client/client.go
var handshakeConfig = goPlugin.HandshakeConfig{
    ProtocolVersion:  1,
    MagicCookieKey:   "ARGO_ROLLOUTS_RPC_PLUGIN",
    MagicCookieValue: "metricprovider",
}

func main() {
    // Configure logging to stderr ONLY -- stdout reserved for go-plugin handshake (DIST-04).
    setupLogging()

    // Serve() prints handshake to stdout, then redirects os.Stdout to a pipe.
    // NOTHING must write to stdout before this line.
    // Phase 2 will add: "RpcMetricProviderPlugin": &rpc.RpcMetricProviderPlugin{Impl: impl}
    goPlugin.Serve(&goPlugin.ServeConfig{
        HandshakeConfig: handshakeConfig,
        Plugins:         map[string]goPlugin.Plugin{},
    })
}

func setupLogging() {
    levelStr := os.Getenv("LOG_LEVEL")
    var level slog.Level
    switch strings.ToLower(levelStr) {
    case "debug":
        level = slog.LevelDebug
    case "warn", "warning":
        level = slog.LevelWarn
    case "error":
        level = slog.LevelError
    default:
        level = slog.LevelInfo
    }
    handler := slog.NewJSONHandler(os.Stderr, &slog.HandlerOptions{Level: level})
    slog.SetDefault(slog.New(handler))
}
```

**Step 1: Create `Makefile`**

Per D-16: targets `build`, `test`, `lint`, `clean`. Plus `lint-stdout` per VALIDATION.md Wave 0.

```makefile
MODULE := github.com/jmichalek132/argo-rollouts-k6-plugin
VERSION ?= $(shell git describe --tags --always --dirty 2>/dev/null || echo "dev")
LDFLAGS := -s -w -X main.version=$(VERSION)

.PHONY: build build-metric build-step test lint lint-stdout clean

build: build-metric build-step

build-metric:
	CGO_ENABLED=0 go build -ldflags "$(LDFLAGS)" -o bin/metric-plugin ./cmd/metric-plugin

build-step:
	CGO_ENABLED=0 go build -ldflags "$(LDFLAGS)" -o bin/step-plugin ./cmd/step-plugin

test:
	go test -race -v -count=1 ./...

lint: lint-stdout
	golangci-lint run

lint-stdout:
	@echo "Checking for stdout usage in non-test code..."
	@if grep -rn 'fmt\.Print\|fmt\.Fprint.*os\.Stdout\|os\.Stdout' cmd/ internal/ --include='*.go' | grep -v '_test.go' | grep -v '// stdout-ok'; then \
		echo "ERROR: found stdout usage in non-test code (stdout reserved for go-plugin handshake)"; \
		exit 1; \
	fi
	@echo "No stdout usage found -- OK"

clean:
	rm -rf bin/
```

Key details:
- `CGO_ENABLED=0` only on build targets, NOT on test (research finding: race detector requires CGO)
- `-ldflags "-s -w"` strips debug info, reduces binary size ~30%
- `lint-stdout` greps for `fmt.Print*`, `os.Stdout` in non-test Go files under cmd/ and internal/
- `lint` depends on `lint-stdout` so stdout check always runs with lint
- `test` uses `-race -v -count=1` for thorough testing

**Step 3: Create `.golangci.yml`**

Per research: golangci-lint v2 uses `version: "2"`, `linters.default: standard`, and `forbidigo` for stdout detection.

```yaml
version: "2"

linters:
  default: standard
  enable:
    - bodyclose
    - noctx
    - exhaustive
    - misspell
    - goconst
    - unconvert
    - forbidigo
  settings:
    forbidigo:
      forbid:
        - pattern: "fmt\\.Print.*"
          msg: "Use slog (writes to stderr) instead of fmt.Print (writes to stdout). stdout is reserved for go-plugin handshake."
        - pattern: "os\\.Stdout"
          msg: "stdout is reserved for go-plugin handshake. Use os.Stderr for output."

formatters:
  enable:
    - gofmt
    - goimports

run:
  go: "1.24"
  timeout: 5m

issues:
  max-issues-per-linter: 0
  max-same-issues: 0
```
