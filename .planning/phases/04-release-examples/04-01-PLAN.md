1. Create `.github/workflows/` directory.

2. Create `.github/workflows/ci.yml` -- CI pipeline that runs on PR and main push (per D-09):

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:

permissions:
  contents: read

jobs:
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-go@v5
        with:
          go-version-file: go.mod
      - uses: golangci/golangci-lint-action@v9
        with:
          version: v2.1.6

  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-go@v5
        with:
          go-version-file: go.mod
      - run: make test

  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-go@v5
        with:
          go-version-file: go.mod
      - run: make build
```

Three parallel jobs: lint (golangci-lint-action v9 with golangci-lint v2.1.6), test (make test -- go test -race), build (make build -- CGO_ENABLED=0 go build both binaries). No release artifacts on non-tag runs per D-09.

3. Create `.github/workflows/release.yml` -- Release pipeline triggered on v* tag push (per D-10):

```yaml
name: Release

on:
  push:
    tags:
      - "v*"

permissions:
  contents: write

jobs:
  release:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0
      - uses: actions/setup-go@v5
        with:
          go-version-file: go.mod
      - uses: goreleaser/goreleaser-action@v7
        with:
          distribution: goreleaser
          version: "~> v2"
          args: release --clean
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```

Critical: `fetch-depth: 0` is required for goreleaser to determine version from git tags (Pitfall 6 from research). Uses `GITHUB_TOKEN` auto-provided by Actions per D-10.

4. Create `.github/workflows/e2e.yml` -- e2e test pipeline triggered on v* tag push and workflow_dispatch (per D-11). This is a separate workflow because kind cluster overhead is too high for every PR:

```yaml
name: e2e

on:
  push:
    tags:
      - "v*"
  workflow_dispatch:

permissions:
  contents: read

jobs:
  e2e:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-go@v5
        with:
          go-version-file: go.mod
      - name: Build plugin binaries
        run: CGO_ENABLED=0 make build
      - name: Run e2e tests
        run: go test -v -tags=e2e -count=1 ./e2e/...
```

Key points per D-11:
- Triggers on `v*` tag push (same as release) and `workflow_dispatch` (manual trigger for ad-hoc runs)
- NOT triggered on PR or main push -- kind overhead too high
- Builds both plugin binaries first with `CGO_ENABLED=0 make build` (e2e tests need the binaries to load into kind)
- Runs `go test -v -tags=e2e -count=1 ./e2e/...` which compiles and runs only the e2e-tagged tests
- `permissions: contents: read` since this workflow only reads code, no write needed
