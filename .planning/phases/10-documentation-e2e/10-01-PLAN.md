# ClusterRole for the argo-rollouts-k6-plugin k6-operator provider.
# Grants the Argo Rollouts controller permissions to manage k6-operator
# TestRun CRDs, inspect runner pods, read pod logs, and read ConfigMaps.
#
# Bind this ClusterRole to the argo-rollouts ServiceAccount:
#   kubectl create clusterrolebinding argo-rollouts-k6-plugin \
#     --clusterrole=argo-rollouts-k6-plugin \
#     --serviceaccount=argo-rollouts:argo-rollouts
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
  name: argo-rollouts-k6-plugin
rules:
  # TestRun and PrivateLoadZone CRDs (k6-operator)
  - apiGroups: ["k6.io"]
    resources: ["testruns", "privateloadzones"]
    verbs: ["create", "get", "list", "watch", "delete"]
  # Runner pod inspection for exit codes
  - apiGroups: [""]
    resources: ["pods"]
    verbs: ["get", "list"]
  # Pod log reading for handleSummary metric extraction
  - apiGroups: [""]
    resources: ["pods/log"]
    verbs: ["get"]
  # ConfigMap reading for k6 test scripts
  - apiGroups: [""]
    resources: ["configmaps"]
    verbs: ["get"]
```

# Polls k6 threshold results from in-cluster test runs.
#

#

      value: test.js
    - name: namespace
  metrics:
    - name: k6-thresholds
      interval: 15s
      successCondition: "result == 1"
      failureLimit: 0
      provider:

            provider: k6-operator
            configMapRef:
              name: "{{args.script-configmap}}"
              key: "{{args.script-key}}"

# Canary Rollout with k6-operator step plugin.
# Triggers an in-cluster k6 load test at 20% canary weight,
# waits for the test to complete, then promotes to 100%.
#
# Prerequisites:
#   1. k6-operator installed (kubectl apply -f https://raw.githubusercontent.com/grafana/k6-operator/v1.3.2/bundle.yaml)
#   2. RBAC ClusterRole applied (see clusterrole.yaml)
#   3. k6 script ConfigMap created (see configmap-script.yaml)
#   4. Plugin registered in argo-rollouts-config ConfigMap
apiVersion: argoproj.io/v1alpha1
kind: Rollout
metadata:
  name: my-app
spec:
  replicas: 3
  revisionHistoryLimit: 3
  selector:
    matchLabels:
      app: my-app
  template:
    metadata:
      labels:
        app: my-app
    spec:
      containers:
        - name: my-app
          image: nginx:1.27
          ports:
            - containerPort: 80
  strategy:
    canary:
      steps:
        - setWeight: 20
        - plugin:
            name: jmichalek132/k6-step
            config:
              provider: k6-operator
              configMapRef:
                name: k6-load-test
                key: test.js
              timeout: "5m"
        - setWeight: 100
```

**rollout-metric.yaml** (per D-03, DOCS-02 -- metric plugin Rollout with analysis step):
```yaml
# Canary Rollout with k6-operator metric plugin analysis gate.
# At 20% canary weight, runs a background AnalysisRun that triggers
# an in-cluster k6 test and evaluates threshold results.
#
# Flow:
#   1. Route 20% traffic to canary
#   2. AnalysisRun triggers k6-operator TestRun via metric plugin
#   3. Metric plugin polls until test completes, evaluates thresholds
#   4. If thresholds pass (result == 1), promote to 100%
#   5. If thresholds fail, automatic rollback
#
# Prerequisites:
#   1. k6-operator installed
#   2. RBAC ClusterRole applied (see clusterrole.yaml)
#   3. k6 script ConfigMap created (see configmap-script.yaml)
#   4. AnalysisTemplate applied (see analysistemplate.yaml)
apiVersion: argoproj.io/v1alpha1
kind: Rollout
metadata:
  name: my-app
spec:
  replicas: 3
  revisionHistoryLimit: 3
  selector:
    matchLabels:
      app: my-app
  template:
    metadata:
      labels:
        app: my-app
    spec:
      containers:
        - name: my-app
          image: nginx:1.27
          ports:
            - containerPort: 80
  strategy:
    canary:
      steps:
        - setWeight: 20
        - analysis:
            templates:
              - templateName: k6-operator-threshold-check
            args:
              - name: script-configmap
                value: k6-load-test
              - name: script-key
                value: test.js
              - name: namespace
                value: default
        - setWeight: 100
```

**configmap-script.yaml** (per D-04):
The k6 script MUST include `export function handleSummary(data) { return { stdout: JSON.stringify(data) } }` -- without it, the metric plugin cannot extract metrics from pod logs. Uses `iterations: 10` (not `duration`) for deterministic completion. Targets a cluster-internal service via `__ENV.TARGET_URL`.
```yaml
# k6 test script for in-cluster execution via k6-operator.
# The script runs HTTP requests against a target service and exports
# handleSummary JSON to stdout for metric extraction by the plugin.
#
# IMPORTANT: The handleSummary function is required for the metric plugin
# to extract detailed metrics (http_req_failed, http_req_duration, etc.)
# from runner pod logs. Without it, only threshold pass/fail is available.
#
# Replace TARGET_URL with your application's cluster-internal service address.
apiVersion: v1
kind: ConfigMap
metadata:
  name: k6-load-test
data:
  test.js: |
    import http from "k6/http";
    import { check } from "k6";

    export const options = {
      vus: 1,
      iterations: 10,
      thresholds: {
        http_req_failed: ["rate<0.01"],
        http_req_duration: ["p(95)<500"],
      },
    };

    // TARGET_URL is injected via k6 runner pod environment variable.
    // Default targets a service in the default namespace.
    const TARGET_URL = __ENV.TARGET_URL || "http://my-app.default.svc.cluster.local";

    export default function () {
      const res = http.get(TARGET_URL);
      check(res, { "status is 200": (r) => r.status === 200 });
    }

    // handleSummary writes k6 summary data as JSON to stdout.
    // The argo-rollouts-k6-plugin reads this from runner pod logs
    // to extract http_req_failed, http_req_duration, and http_reqs metrics.
    export function handleSummary(data) {
      return { stdout: JSON.stringify(data) };
    }
```
